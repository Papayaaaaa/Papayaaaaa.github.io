# Failover command for streaming replication.
# This script assumes that DB node 0 is primary, and 1 is standby.
#
# If standby goes down, do nothing. If primary goes down, create a
# trigger file so that standby takes over primary node.
#
# Arguments: $1: failed node id. $2: new master hostname. $3: path to
# trigger file.
failed_node=192.168.1.101
new_master=192.168.1.102
trigger_file=$3
# Do nothing if standby goes down.
if~~ [ ~~$failed_node~~ = 1 ]; ~~then
exit~~ 0;~~
fi
# Create the trigger file.
/usr/bin/ssh -T ~~$new_master~~ /bin/touch ~~$trigger_file~~
exit~~ 0;~~

####################### failover_stream.sh ############################
#!/bin/bash -x
falling_node=$1 # %d
old_primary=192.168.1.101 # %P
new_primary=192.168.1.102 # %H
pgdata=/app/pgsql/data/ # %R

pghome=/usr/pgsql-9.5
log=/opt/pgpool/failover.log

date >> $log
echo “failed_node_id=$falling_node new_primary=$new_primary” >> $log

if [ $falling_node = $old_primary ]; then
if [ $UID -eq 0 ]
then
su postgres -c “ssh -T postgres@$new_primary $pghome/bin/pg_ctl promote -D $pgdata”
else
ssh -T postgres@$new_primary $pghome/bin/pg_ctl promote -D $pgdata
fi
exit 0;
fi;
exit 0;

Github/Gitee+hexo搭建博客

发表于 2019-03-29
  1. 注册Github/Gitee账号，安装Git客户端，生成并配置GitHub公钥，截图步骤略（这一步是为了让后面推送时不需要每次都输入账号密码）
  2. 创建仓库并在设置中开启Git Pages功能
    最好两者仓库的名字都相同，Github Page要求仓库名字为 [账号名].github.io，Gitee无要求但不可以以.git结尾
    例如我的github帐号名为Papayaaaaa
    则Github Page 仓库为名为 Papayaaaaa.github.io Gitee仓库名为Papayaaaaa
    Github(Pages配置在settings页面最底部，默认配置就ok) Gitee
  3. 安装Node.js，网址略，安装完成后查看nodejs版本及npm版本验证安装是否完成
  4. 安装Hexo，在电脑中创建一个新文件夹，命名为Papayaaaaa.github.io作为博客文件夹
    看到有博客说名称和仓库名不同会有问题但是理论上不同名也应该ok的，不想麻烦的话建议还是同名
  5. 在文件夹中打开cmd（按住Shift右键）
    使用npm命令安装Hexo
    npm install -g hexo-cli
  6. 安装Git插件 cmd输入
    npm install hexo-deployer-git --save
  7. 初始化博客，cmd输入
    hexo init blog
  8. 创建一个新页面， cmd输入
    hexo new "My first blog"
    这句命令即创建一个新的名为My first blog的页面，页面以.md文件储存在/source/_post/文件夹下，页面语言为Markdown
  9. 生成静态博客页面 hexo g g即generate，生成博客
  10. 本地测试博客框架 hexo s s即server， 可以在localhost:4000查看本地博客框架
  11. 将本地博客推送到Github/Gitee
    修改站点根目录下的站点配置文件_config.yml，记事本或Notepad打开
    修改最下面一行 文件最开头可以设置博客标题等 修改完保存文件，本地博客根目录下命令行
    hexo clean
    hexo g
    hexo d##即deploy，进行远程推送
  12. 每次自己在本地写好博客都可以通过这三个命令进行页面生成与推送
  13. 貌似推送到gitee需要在source文件夹下建立一个Staticfile文件否则无法识别静态网页（？不知道为啥）

遇到的一些问题：
之前在Gitee上的仓库名称和我账号名称不一样，推送之后在Gitee上就无法显示出页面，设置_config.yml的url和root属性可以解决但设置之后Github上又显示不出来框架只显示静态页面，搞了半天也不知都为啥，最后只能删了重来大家都统一命名，猜测是命名不同会导致路径的一些变化，dbq本人真的是纯正的半桶水orz

PS：
码云（Gitee）上推送后无法自动部署网站，每次推送后需要自己手动部署（自动部署需要购买VIP 99r/year），其实用Gitee本来是觉得会快一点但是好像也没有快很多的样子orz.

PPS：
绑定域名就按网上来就ok，但是注意直接在仓库根目录建立CNAME文件在第二次推送的时候会被覆盖掉因此需要在本地目录/source文件夹下建立CNAME文件。
